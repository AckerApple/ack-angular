Prompts

Modal driven prompts to require user confirmation

This component requires the element <ack-fixed-element-stage> to be present somewhere in your app
alert confirm

Usage Example

import { Prompts } from "./ack-angular"

class MyClass{
  constructor(public prompts:Prompts){}

  runConfirm():EventEmitter<boolean>{
    return this.prompts.confirm('This is a confirm example')
    .subscribe(result=>console.log('result of confirm:' +result))
  }

  runAlert():EventEmitter<boolean>{
    return this.prompts.alert('This is an alert example')
    .subscribe(result=>console.log('alert prompt closed'))
  }
}

Usage Example

import { AckApi } from "ack-angular"
import { Component } from "@angular/core"

@Component({...})
export class SomeClass{
  offlineModel = {
    name:'ackHttpTests',
    maxAge:15000,//GET cache busted at 15 seconds
    maxTry:50//POST failure retries allowed
  }

  constructor(public AckApi:AckApi){}

  monitorOnlineStatus(){
    window.addEventListener('online',()=>{
      if(navigator.onLine){
        this.AckApi.processQue('ackHttpTests')
      }
    })
  }

  clearPostQue(){
    return this.AckApi.clearQue('ackHttpTests')
  }

  clearGetCache(){
    return this.AckApi.clearCache('ackHttpTests')
  }

  request(){
    return this.AckApi.request({
      url:'...'.
      method:'...'
      offlineModel:this.offlineModel
    })
    .then( response=>console.log('200',response) )
    .catch(e=>{

      //ignore error, has been qued for when internet returns
      if(e.sendFailMeta)return

      return Promise.reject(e)
    })
  }

  POST(){
    return this.AckApi.post('url', {data}, {offlineModel:this.offlineModel})
    .then( response=>console.log('200',response) )
    .catch(e=>{
      //ignore error, has been qued for when internet returns
      if(e.sendFailMeta)return

      return Promise.reject(e)
    })
  }

  GET(){
    return this.AckApi.get('url', {offlineModel:this.offlineModel})
    .then( response=>console.log('200',response) )
    .catch( e=>console.error(e) )
  }
}

import { AckOffline } from "ack-angular"

@Component({...})

    .catch(e=>console.error(e))

  clear(){

  }

AckQue

Offline storage used to provide association between data and data-processors

view details
Que Test
{{i}}
add
Completed {{i}} :
TRY THIS
  • Put items into que
  • Disable your internet connection (unplug it)
  • Enable your internet connection
  • The entire que will be processed (takes seconds)
You are currently : {{statusOnlineModel ? 'online' : 'offline'}}

Usage Example

import { AckOffline } from "ack-angular"
import { Component } from "@angular/core"

@Component({...})
export class SomeClass{
  constructor(public AckOffline:AckOffline){
    this.AckOffline.registerQueHandler('postHttpQue',quedData=>{
      console.log('browser back online, ready to transmit data', quedData)
    })

    this.onlineEvent = ()=>this.onOnline()

    window.addEventListener('online',this.onlineEvent)
  }

  /** anytime we come back from offline, lets process our que */
  onOnline(){
    this.AckOffline.processQue()
    .catch(e=>console.error(e))
  }

  set(data){
    return this.AckOffline.set('my-app-que', data)
    .catch(e=>console.error(e))
  }
  
  ngOnDestroy(){
    window.removeEventListender(this.onlineEvent)
  }
}
